196 
197  if( norm == std::string("L2") )
198  {
199  error = exact_sol.l2_error(system_name, var);

204  }

208  exit(1);

217  << "norm of error = " << error << std::endl
218  << "norm type = " << norm << std::endl
219  << "var = " << var << std::endl;
220  }
221  else

223  std::cout << "PASSED!" << std::endl
